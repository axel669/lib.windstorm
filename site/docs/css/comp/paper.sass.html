Paper

Generic container component for making areas visually distinct from each other on the page.

Wind Functions

  • @outline
  • $color

Areas

  • slot[content] required
  • slot[header]
  • slot[footer]
<ws-paper>
    <ws-flex ws-x="slot[content]">
        Just some paper
    </ws-flex>
</ws-paper>

<ws-paper>
    <ws-titlebar ws-x="slot[header]">
        <span ws-x="slot[title]">This one has a header</span>
    </ws-titlebar>
    <ws-flex ws-x="slot[content]">
        With some content \o/
    </ws-flex>
</ws-paper>

<ws-paper ws-x="@outline $color[secondary]">
    <ws-titlebar ws-x="slot[header]">
        <span ws-x="slot[title]">Give them some color</span>
    </ws-titlebar>
    <ws-flex ws-x="slot[content]">
        And a border + some controls in the footer
    </ws-flex>
    <ws-grid ws-x="gr-col[1fr_1fr]" slot="footer">
        <button ws-x="@flat $color[danger]">Cancel</button>
        <button ws-x="@flat $color[secondary]">Use Windstorm</button>
    </ws-grid>
</ws-paper>
